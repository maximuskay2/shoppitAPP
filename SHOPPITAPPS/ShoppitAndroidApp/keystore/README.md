# Keystore Configuration for APK Signing

# Place your keystore file in this directory.
# Supply the following details as environment variables when building:
#   KEYSTORE_PATH: Path to your keystore file (default: ./keystore/release.keystore)
#   KEYSTORE_PASSWORD: Password for the keystore
#   KEY_ALIAS: Alias for the key
#   KEY_PASSWORD: Password for the key

# Example usage:
# export KEYSTORE_PATH=./keystore/release.keystore
# export KEYSTORE_PASSWORD=your_keystore_password
# export KEY_ALIAS=your_key_alias
# export KEY_PASSWORD=your_key_password
# ./gradlew assembleRelease
